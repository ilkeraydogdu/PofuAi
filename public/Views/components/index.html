Component Kullanımı

PofuAI componentlerini kullanmak için iki adım izleyin:

1. Controller'da Component Verisi Hazırlama
from core.Services.UIService import UIService

def render_page():
    ui = UIService()
    
    # Component verisini hazırla
    alert_data = ui.create_alert(
        message="İşlem başarıyla tamamlandı",
        type="success",
        dismissible=True
    )
    
    # Sayfayı render et
    return render_template('page.html', alert=alert_data)
