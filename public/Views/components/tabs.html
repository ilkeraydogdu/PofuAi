Alternatif olarak, UI Service'i kullanarak tabs verisi oluşturabilirsiniz:

{% raw %}from core.services.ui_service import get_ui_service

# UI Service'den tabs verisi oluştur
ui_service = get_ui_service()
tabs_data = ui_service.create_tabs(
    items=[
        {'id': 'home', 'title': 'Ana Sayfa', 'content': 'Ana sayfa içeriği'},
        {'id': 'profile', 'title': 'Profil', 'content': 'Profil içeriği'},
        {'id': 'messages', 'title': 'Mesajlar', 'content': 'Mesajlar içeriği'}
    ],
    active_tab='home'
)

# Template'e aktar
return render_template('my_page.html', tabs=tabs_data){% endraw %}
